100  inline Vector<unsigned> arrange_architecture(void) const
101  {
102  const unsigned layers_number = get_layers_number();
103 
104  Vector<unsigned> architecture;
105 
106  if(layers_number != 0)
107  {
108  const unsigned inputs_number = get_inputs_number();
109  architecture.resize(1+layers_number);
110 
111  architecture[0] = inputs_number;
112 
113  Vector<unsigned> layers_size = arrange_layers_perceptrons_numbers();
114 
115  for(unsigned i = 0; i < layers_number; i++)
116  {
117  architecture[i+1] = layers_size[i];
118  }
119  }
120 
121  return(architecture);
122  }

126  inline Vector<unsigned> arrange_complexity(void) const
127  {

130  Vector<unsigned> complexity;

132  if(layers_number != 1)

134  complexity.resize(layers_number-1);

137

139  {
140  complexity[i] = layers_size[i];
141  }
142  }
143 
144  return(complexity);
