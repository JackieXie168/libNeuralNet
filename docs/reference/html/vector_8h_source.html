814 Histogram<Type> calculate_histogram(const unsigned& bins_number = 10) const

816  // Control sentence (if debug)

818  #ifndef NDEBUG

828  throw std::logic_error(buffer.str());
829  }
830

832 
833  Vector<Type> minimums(bins_number);
834  Vector<Type> maximums(bins_number);
835 
836  Vector<Type> centers(bins_number);
837  Vector<unsigned> frequencies(bins_number, 0);
838 
839  const Vector<Type> minimum_maximum = calculate_minimum_maximum();
840 
841  const Type minimum = minimum_maximum[0];
842  const Type maximum = minimum_maximum[1];
843 
844  const double length = (maximum-minimum)/(double)bins_number;
845 
846  minimums[0] = minimum;
847  maximums[0] = minimum + length;
848  centers[0] = (maximums[0] + minimums[0])/2.0;
849 
850  // Calculate bins center
851 
852  for(unsigned i = 1; i < bins_number; i++)
853  {
854  minimums[i] = minimums[i-1] + length;
855  maximums[i] = maximums[i-1] + length;
856 
857  centers[i] = (maximums[i] + minimums[i])/2.0;
858  }
859 
860  // Calculate bins frequency
861 
862  const unsigned this_size = this->size();
863 
864  for(unsigned i = 0; i < this_size; i++)
865  {
866  if((*this)[i] < centers[0])
867  {
868  frequencies[0]++;
869  }
870 
871  for(unsigned j = 1; j < bins_number-1; j++)
872  {
873  if((*this)[i] >= minimums[j] && (*this)[i] < maximums[j])
874  {
875  frequencies[j]++;
876  }
877  }
878 
879  if((*this)[i] >= centers[bins_number-1])
880  {
881  frequencies[bins_number-1]++;
882  }
883  }
884 
885  Histogram<Type> histogram(bins_number);
886  histogram.centers = centers;
887  histogram.frequencies = frequencies;
888
