95  static double calculate_trapezoid_integral(const T& t,
96  double (T::*f)(const double&) const , const double& a, const double& b, const unsigned& n)
97  {
98  // Integration step
99 
100  const double h = (b-a)/(n-1.0);
101 
102  // Sum
103 
104  double sum = (t.*f)(a)/2.0;
105 
106  for(unsigned i = 1; i < n-1; i++)
107  {
108  sum += (t.*f)(a + i*h);
109  }
110 
111  sum += (t.*f)(b)/2.0;
112 
113  // Trapezoidal rule
114

166  static double calculate_Simpson_integral(const T& t, double (T::*f)(const double&) const , const double& a, const double& b, const unsigned& n)
167  {
168  const double h = (b-a)/(n-1.0);
169 
170  double sum = (t.*f)(a)/3.0;
171 
172  for(unsigned i = 1; i < n-1; i++)
173  {
174  if(i%2 != 0) // odd
175  {
176  sum += 4.0*(t.*f)(a + i*h)/3.0;
177  }
178  else // even

180  sum += 2.0*(t.*f)(a + i*h)/3.0;

182  }

185 
186  return(h*sum);
